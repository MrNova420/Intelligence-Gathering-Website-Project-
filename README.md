```python
import requests

# Submit intelligence query
response = requests.post('https://your-domain.com/api/v1/queries/', 
    headers={'Authorization': 'Bearer YOUR_TOKEN'},
    json={
        'query_type': 'email',
        'query_value': 'target@example.com',
```
